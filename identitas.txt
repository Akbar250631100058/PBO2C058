class Person {
    String name;
    int age;
    String city;

    void displayInfo() {
        System.out.println("Name: " + name);
        System.out.println("Age: " + age);
        System.out.println("City: " + city);
    }
}

public class Main {
    public static void main(String[] args) {
        Person p1 = new Person();

        p1.name = "Bahtiar";
        p1.age = 18;
        p1.city = "Pamekasan";

        p1.displayInfo();
    }
}
